#2 - Intalling/Decompressing Packages
    
    #2.1 To install/decompress rpm file (*.rpm)
     +Installing
        $ rpm -ivh rpmFilename.rpm
     +Upgrading
        $ rpm -Uvh rpmFilename.rpm
     +Uninstall (remove)
        $ rpm -ev
     +List of recently installed rpms
        $ rpm -qa --last
    
    #2.2 To decompress a tar file (*.tar)
     $ tar -xvf tarFileName.tar [specific directory/file]
        + -x is for extracting
        + -v is for verbose (printing nicely)
        + -f is for specifying a filename
        + -j is for decompressing *.tar.bz2 extension
        + -z is for decompressing *.tar.gz extension

    #2.3 How to install tarball files (*.tar.gz)
        +Decompress tar file
         $ tar -xvf tarFileName.tar.gz (See rule above -decompress tar file)
    
        +Go to extracted directory
    
        +Configurate and Make
         $ ./configure #to check all files and programs are OK
         $ make
         $ make install
